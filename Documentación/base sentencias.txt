CREATE OR REPLACE FUNCTION public.f_botones(
	id_param integer)
    RETURNS character varying
    LANGUAGE 'plpgsql'

    COST 100
    VOLATILE 
AS $BODY$
DECLARE
	v_nombre varchar;
BEGIN
	select nombre into v_nombre
	from botones
	where id = id_param;
	return v_nombre;
END
$BODY$;

ALTER FUNCTION public.f_botones(integer)
    OWNER TO postgres;

CREATE OR REPLACE FUNCTION public.f_clientes(
	id_param integer)
    RETURNS character varying
    LANGUAGE 'plpgsql'

    COST 100
    VOLATILE 
AS $BODY$
DECLARE
	v_nombre varchar;
BEGIN
	select (apellido || ' ' || nombre) into v_nombre
	from clientes
	where id = id_param;
	return v_nombre;
END
$BODY$;

ALTER FUNCTION public.f_clientes(integer)
    OWNER TO postgres;
